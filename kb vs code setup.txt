CREATE TABLE AllRecipes (
	RecipeId int NOT NULL AUTO_INCREMENT,
	Name varchar(255),
	Url varchar(255), #this might need to be increased
	PRIMARY KEY (RecipeId)
);

CREATE TABLE FavRecipes (
	CustomerId int NOT NULL,
	RecipeId int NOT NULL,
	FOREIGN KEY (CustomerId) REFERENCES CustomerInfo(CustomerId),
	FOREIGN KEY (RecipeId) REFERENCES AllRecipes(RecipeId),
	CONSTRAINT Relationship UNIQUE (CustomerId, FriendId)
);

CREATE TABLE KitchenConstraints (
	CustomerId int NOT NULL,
	Burners int, #number of stovetop burners
	Pans int,
	Pots int,
	CuttingBoards int,
	Knives int,
	PRIMARY KEY (CustomerId),
	FOREIGN KEY (CustomerId) REFERENCES CustomerInfo(CustomerId)
);

CREATE TABLE FriendsList (
	CustomerId int NOT NULL,
	FriendId int NOT NULL,
	FOREIGN KEY (CustomerId) REFERENCES CustomerInfo(CustomerId),
	FOREIGN KEY (FriendId) REFERENCES CustomerInfo(CustomerId), # both ids should be found in customer list
	CONSTRAINT Relationship UNIQUE (CustomerId, FriendId)
